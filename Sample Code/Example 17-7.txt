	ORG 0H
MAIN:
	CLR P1.0				
	SETB P2.7
MONITOR:
	JNB P2.7, FIFTYPERCENT
	SETB P1.0				
	MOV R5, #25
	ACALL DELAY
	CLR P1.0				
	MOV R5, #75
	ACALL DELAY
	SJMP MONITOR			
FIFTYPERCENT:
	SETB P1.0				
	MOV R5, #50
	ACALL DELAY
	CLR P1.0				
	MOV R5, #50
	ACALL DELAY
	SJMP MONITOR
DELAY:
H1:	MOV R2, #100
H2:	MOV R3, #255
H3:	DJNZ R3, H3
	DJNZ R2, H2
	DJNZ R5, H1
	RET
	END
