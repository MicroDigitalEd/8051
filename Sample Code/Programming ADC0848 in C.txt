#include <reg51.h>
sbit CS = P2^4;
sbit WR = P2^5;
sbit RD = P2^6;
sbit INTR = P2^7;
void main()
  {
    unsigned char value;
    INTR = 1;         	
    CS = 1;            	
    WR = 1;
    RD = 1;
    while(1)
      {
        P1 = 0x0A;  		
        CS = 0;       	
        WR = 0;      	
        Delay();      	
        WR = 1; 		
        CS = 1;			
        P1 = 0xFF;    	
        while(INTR==1);  	
        CS = 0;        	
        RD = 0;       	
        Delay();
        RD = 1;			
        value = P1;    	
        CS = 1;
        ConvertAndDisplay(value);  
      }
  }
