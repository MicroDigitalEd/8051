DATA_ADDR	EQU	400H 		
COUNT		EQU	5    		
RAM_ADDR	EQU	40H 		
	ORG	0
	ACALL	COPY_1 		
	MOV	A,0C4H  		
	SETB ACC.0			
	MOV	0C4H,A		
	ACALL	COPY_2		
	MOV	TMOD,#20H		
	MOV	TH1,#-3		
	MOV	SCON,#50H
	SETB	TR1
	ACALL	COPY_COM		
	SJMP	$      		
COPY_1:		
	MOV	DPTR,#DATA_ADDR      
	MOV	R0,#RAM_ADDR
	MOV	R2,#COUNT
H1:	CLR	A
	MOVC	A,@A+DPTR
	MOV	@R0,A
	INC	DPTR
	INC	R0
	DJNZ	R2,H1
	RET		
COPY_2:		
	MOV	DPTR,#0		
	MOV	R0,#RAM_ADDR
	MOV	R2,#COUNT
H2:	MOV	A,@R0   		
	MOVX	@DPTR,A		
	INC	DPTR
	INC	R0
	DJNZ	R2,H2
	RET		
COPY_COM:		
		MOV	DPTR,#0		
		MOV	R2,#COUNT
H3:	MOVX	A,@DPTR 		
		ACALL SERIAL 		
		INC	DPTR
		DJNZ	R2,H3
		RET		
SERIAL:
		MOV	SBUF,A
H4:	JNB	TI,H4
		CLR	TI
		RET
		ORG	400H
MYBYTE:	DB	"HELLO"
		END
