#include <reg51.h>
sbit mybit=P1^5;
sbit SW=P1^7;
void T0M1Delay(unsiged char);
void main(void)
  {
    SW=1;				
    while(1)
      {
        mybit=~mybit;		
        if(SW==0)		
          T0M1Delay(0);
        else
          T0M1Delay(1);
      }
  }
void T0M1Delay(unsigned char c)
  {
    TMOD=0x01;
    if(c==0)
      {
        TL0=0x67;		
        TH0=0xFC;
      }
    else
      {
        TL0=0x9A;		
        TH0=0xFD;
      }
    TR0=1;
    while(TF0==0);
    TR0=0;
    TF0=0;
  }
