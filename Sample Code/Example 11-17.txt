#include <reg51.h>
sbit WAVE = P2^1;
unsigned char cnt;
void timer0() interrupt 1
  {
    WAVE = ~WAVE;		
  }
void timer1() interrupt 3
  {
    cnt++;				
    P0 = cnt;			
  }
void main()
  {
    cnt = 0;			
    TMOD = 0x42;
    TH0 = 0x-46;		
    IE = 0x86;			
    TR0 = 1;			
    TR1 = 1;			
    while(1);			
  }
