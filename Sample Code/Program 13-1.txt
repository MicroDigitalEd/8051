	RD	BIT P2.5		
	WR	BIT P2.6		
	INTR	BIT P2.7		
	MYDATA	EQU P1  	
	MOV	P1,#0FFH		
	SETB	INTR
BACK:	CLR	WR     		
	SETB	WR     		
HERE:	JB	INTR,HERE		
	CLR	RD      		
	MOV	A,MYDATA    	
	ACALL	CONVERSION 		
	ACALL	DATA_DISPLAY 	
	SETB	RD    		
	SJMP	BACK
CONVERSION:
	MOV	B,#10
	DIV	AB
	MOV	R7,B		
	MOV	B,#10
	DIV	AB
	MOV	R6,B
	MOV	R5,A		
	RET
DATA_DISPLAY
	MOV	P0,R7
	ACALL	DELAY
	MOV	P0,R6
	ACALL	DELAY
	MOV	P0,R5
	ACALL	DELAY
	RET
