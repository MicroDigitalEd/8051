#include <reg51.h>
sbit WAVE = P0^1;
void timer0() interrupt 1
  {
    WAVE = ~WAVE;		
  }
void serial0() interrupt 4
  {
    if(TI == 1)
      {
        TI = 0;			
      }
    else
      {
        P0 = SBUF;		
        RI = 0;			
      }
  }
void main()
  {
    unsigned char x;
    P1 = 0xFF;			
    TMOD = 0x22;
    TH1 = 0xF6;			
    SCON = 0x50;
    TH0 = 0xA4;			
    IE = 0x92;			
    TR1 = 1;			
    TR0 = 1;			
    while(1)
      {
        x = P1;			
        SBUF = x;		
        P2 = x;			
      }
  }
