    SCON = 0x50;
    TR1  = 1;

      {
        XBYTE[0x0+x] = msg[x];
      }
    for(x=0; x<5;x++)	 
      { 
        SBUF = XBYTE[0x0+x];
        while(TI==0);
        TI=0;
      }
    while(1); 				
}
