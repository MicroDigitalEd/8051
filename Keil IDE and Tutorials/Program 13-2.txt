#include <reg51.h>
sbit RD = P2^5;
sbit WR = P2^6;
sbit INTR = P2^7;
sfr MYDATA = P1;		
void ConvertAndDisplay(unsigned char value);
void MSDelay(unsigned int value);
void main()
  {
    MYDATA = 0xFF;		
    INTR = 1;			
    RD = 1;			
    WR = 1;			
    while(1)
     {
       WR = 0;			
       WR = 1;
       while(INTR == 1);	
       RD = 0;			
       value = MYDATA;	
       ConvertAndDisplay(value);
       RD = 1;
     }
  }
void ConvertAndDisplay(unsigned char value)
  {
    unsigned char x,d1,d2,d3;
    x=value/10;
    d1=value%10;
    d2=x%10
    d3=x/10
    P0=d1;				
    MSDelay(250);
    P0=d2;
    MSDelay(250);
    P0=d3;				
    MSDelay(250);
  }
  
void MSDelay(unsigned int value)
  {
    unsigned char x,y;
    for(x=0;x<value;x++)
      for(y=0;y<1275;y++);
  }
